#include <stdio.h>
#include <stdbool.h>;

int BMSearch(char wzorzec[10],char slowo[100])
{
	int wzorzec_dl = strlen(wzorzec) -2;
	int i = wzorzec_dl;
	int stala = 0;
	int licznik_wzorcow = 0;
	//printf("%d\n", wzorzec_dl);
	//printf("%d\n", strlen(slowo));
	while (i < strlen(slowo))
	{
		//printf("\nsprawdzana litera %c dla i = %d\n", slowo[i],i);
		if (stala == 0)
		{
			for (int j = 0; j <= wzorzec_dl; j++)
			{
				//printf("jestem w for:  ");
				//printf("sprawdzana litera %c dla wzorzec_dl - j = %d\n", wzorzec[wzorzec_dl - j], wzorzec_dl - j);
				if (wzorzec[wzorzec_dl - j] == slowo[i])
				{
					i = i + j;
					j = wzorzec_dl + 1;
				}
				printf("i = %d", i);
			}
		}
		if (slowo[i - stala] == wzorzec[wzorzec_dl - stala])
		{
			stala++;
			//printf("%d", stala);
			if (stala == wzorzec_dl)
			{
				licznik_wzorcow++;
				i +=wzorzec_dl+1;
				stala = 0;
			}
		}
		else if (stala > 0)
		{
			i += stala;
			stala = 0;
		}
		else
		{
			stala = 0;
			i += wzorzec_dl+1;
		}
		
	}
	return licznik_wzorcow;
}

int main()
{
	char wzorzec[10];
	fgets(wzorzec, sizeof wzorzec, stdin);
	
	char slowo[100];
	fgets(slowo, sizeof slowo, stdin);

	int ilosc_wzorcow = BMSearch(wzorzec, slowo);

	if (ilosc_wzorcow > 0)
	{
		printf("Ilosc wzorcow w tekscie: %d", ilosc_wzorcow);
	}
	else printf("Brak wzorcow w tekscie");
	return 0;
}
